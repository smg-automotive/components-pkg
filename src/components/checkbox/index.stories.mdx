import { useState } from 'react';

import { Canvas, Meta, Story } from '@storybook/addon-docs';

import { Stack } from '@chakra-ui/react';

import Checkbox from './index.tsx';

<Meta title="Components/Checkbox" component={Checkbox} />

export const Template = ({ ...args, value, onChange }) => {
  const [currentValue, setCurrentValue] = useState(value);
  const onChangeHandler = onChange ? action('onChange') : () => {};
  return (
    <Checkbox
      {...args}
      onChange={
        value
          ? (e) => {
              setCurrentValue(e.target.value);
              onChangeHandler(e);
            }
          : onChangeHandler
      }
      value={value ? currentValue : undefined}
    />
  );
};
