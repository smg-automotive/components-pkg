<Meta title="Components/Filters/Checkbox" component={CheckboxFilter} />

export const Template = ({ selected, ...args }) => {
  const [checked, setChecked] = useState(selected || []);
  return (
    <CheckboxFilter
      {...args}
      selected={checked}
      apply={(filters) => {
        setChecked(filters[args.name]);
      }}
      options={[
        { label: 'New', value: 1 },
        { label: 'Used', value: 2 },
        { label: 'Old-timer', value: 3 },
      ]}
    />
  );
};
