export const Slide = ({ index }) => (
  <Flex
    justifyContent="center"
    alignItems="center"
    height="600px"
    position="relative"
  >
    <img
      src={images[index]}
      alt="a great kitten"
      width="100%"
      height="100%"
      style={{
        objectFit: 'cover',
      }}
    />
    <Box position="absolute" backgroundColor="gray.400" p="md" color="white">
      Slide {index}
    </Box>
  </Flex>
);

export const Template = (args) => (
  <FullHeight>
    <Carousel
      {...args}
      onSlideClick={args.onSlideClick ? action('onSlideClick') : undefined}
    >
      {Array(args.numberOfSlides || 6)
        .fill(0)
        .map((_, i) => (
          <Slide
            index={i + 1}
            key={`slide-${i + 1}`}
          />
        ))}
    </Carousel>
  </FullHeight>
);
