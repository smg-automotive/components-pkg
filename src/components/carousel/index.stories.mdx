import { Canvas, Meta, Story } from '@storybook/addon-docs';
import { action } from '@storybook/addon-actions';

import Carousel from './index.tsx';
import Flex from '../flex';
import FullHeight from '../fullHeight';

<Meta title="Components/Data display/Carousel" component={Carousel} />

export const Slide = ({ index }) => (
  <Flex
    backgroundColor={['red.500', 'blue.500'][index % 2]}
    justifyContent="center"
    alignItems="center"
    height="300px"
  >
    Slide {index}
  </Flex>
);

export const Template = (args) => (
  <FullHeight>
    <Carousel
      {...args}
      onSlideClick={args.onSlideClick ? action('onSlideClick') : undefined}
    >
      {Array(6)
        .fill(0)
        .map((_, i) => (
          <Slide index={i + 1} key={`slide-${i + 1}`} />
        ))}
    </Carousel>
  </FullHeight>
);
