import { useState } from 'react';
import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';
import { action } from '@storybook/addon-actions';
import { Box } from '@chakra-ui/react';

import DatePicker from './index.tsx';

<Meta
  title="Forms/Date-picker"
  component={DatePicker}
  parameters={{
    controls: {
      sort: 'alpha',
      expanded: true,
    },
    actions: ['change', 'blur', 'focus'],
  }}
  args={{
    onChange: true,
    onBlur: true,
    onFocus: true,
  }}
/>

export const Container = ({ children }) => {
  return (
    <Box w="100%" maxW="250px">
      {children}
    </Box>
  );
};

export const Template = ({ value, onChange, onBlur, onFocus, ...args }) => {
  const [currentValue, setCurrentValue] = useState(value);
  const onChangeHandler = onChange
    ? action('change')
    : () => {
        return;
      };
  return (
    <Container>
      <DatePicker
        {...args}
        onBlur={onBlur ? action('blur') : undefined}
        onFocus={onFocus ? action('focus') : undefined}
        onChange={(e) => {
          setCurrentValue(e.target.value);
          onChangeHandler(e);
        }}
        value={value ? currentValue : undefined}
        name="datepicker"
      />
    </Container>
  );
};

<Canvas>
  <Story
    name="Date Picker"
    args={{
      size: 'lg',
      min: new Date(),
      value: '',
    }}
    argTypes={{
      size: {
        options: ['md', 'lg'],
        control: 'select',
      },
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
